\newcommand{\HeatmapNode}[1]{

  {
    \renewcommand{\trimmed}{-200}
  }
  % else
  {
    \ifthenelse{#1 > 200}
    {
      \renewcommand{\trimmed}{200}
    }
    % else
    {
      \renewcommand{\trimmed}{#1}
    }
  }

  \ifthenelse{#1 < 0}
  {
    \pgfmathparse{round(\trimmed * -0.5)}
    \node [fill=red!\pgfmathresult] {#1};
  }
  % else
  {
    \pgfmathparse{round(\trimmed * 0.5)}
    \node [fill=blue!\pgfmathresult] {#1};
  }
}
